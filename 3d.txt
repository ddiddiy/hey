using UnityEngine;

public class RangeChecker : MonoBehaviour
{
    [SerializeField] Transform target;
    [SerializeField] float range = 5f;
    [SerializeField] float moveSpeed = 5f;
    bool targetWasInRange = false;
    void Update()
    {
        HandleMovement();
        HandleRangeCheck();
        HandleAngleCheck();
    }
    void HandleMovement()
    {
        float moveX = Input.GetAxis("Horizontal");
        float moveZ = Input.GetAxis("Vertical");

        Vector3 direction = new Vector3(moveX, 0f, moveZ);
        transform.Translate(direction * moveSpeed * Time.deltaTime, Space.World);
    }
    void HandleRangeCheck()
    {
        float distance = Vector3.Distance(transform.position, target.position);

        if (distance <= range && !targetWasInRange)
        {
            Debug.Log("Player entered target range");
            targetWasInRange = true;
        }
        else if (distance > range && targetWasInRange)
        {
            Debug.Log("Player exited target range");
            targetWasInRange = false;
        }
    }

    void HandleAngleCheck()
    {
        Vector3 directionToTarget = target.position - transform.position;
        if (directionToTarget.sqrMagnitude < 0.0001f)
            return;
        directionToTarget.Normalize();
        float dot = Vector3.Dot(transform.forward, directionToTarget);
        dot = Mathf.Clamp(dot, -1f, 1f);
        float angle = Mathf.Acos(dot) * Mathf.Rad2Deg;
        Debug.Log($"Angle to target: {angle:F2}");
    }
}

using UnityEngine;
using UnityEngine.AI;

public class ai_chase_player : MonoBehaviour
{
    public Transform player;
    public float detectionRange = 5.0f;
    private NavMeshAgent agent;
    void Start()
    {
        agent = GetComponent<NavMeshAgent>();
    }
    void Update()
    {
        if (player == null) return;

        float distanceToPlayer = Vector3.Distance(
            transform.position,
            player.position
        );
        if (distanceToPlayer < detectionRange)
        {
            agent.SetDestination(player.position);
            Debug.Log("AI is chasing the player");
            if (distanceToPlayer < 1f)
            {
                Debug.Log("Enemy Wins!");
                StopGame();
            }
        }
        else
        {
            agent.ResetPath();
        }
    }
    void StopGame()
    {
        Debug.Log("Game Over. The Enemy caught you.");

#if UNITY_EDITOR
        UnityEditor.EditorApplication.isPlaying = false;
#else
        Application.Quit();
#endif
    }
}
