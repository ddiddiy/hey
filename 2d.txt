script > movement:
using UnityEngine;
public class movement : MonoBehaviour
{
    public float speed = 5f;
    private Rigidbody2D rb;
    void Start()
    {
        rb = GetComponent<Rigidbody2D>();
    }
    void Update()
    {
        float movex = Input.GetAxis("Horizontal");
        float movey = Input.GetAxis("Vertical");
        Vector2 movement = new Vector2(movex, movey);
        rb.linearVelocity = movement * speed;
    }
}

attach file to square > run

public class scaling : MonoBehaviour
{
    public float speed = 5f;
    void Update()
    {
        float change = Input.GetAxis("Vertical");
        transform.localScale += Vector3.one * change * speed * Time.deltaTime;

using UnityEngine;

public class jump_move_color : MonoBehaviour
{
    public float speed = 5f;
    public float jumpForce = 10f;
    public Color left = Color.red;
    public Color right = Color.blue;
    public Color idle = Color.white;
    private Rigidbody2D rb;
    private SpriteRenderer sr;
    void Start()
    {
        rb = GetComponent<Rigidbody2D>();
        sr = GetComponent<SpriteRenderer>();
    }
    void Update()
    {
        float move = Input.GetAxis("Horizontal");
        rb.linearVelocity = new Vector2(move*speed, rb.linearVelocity.y);
        if(Input.GetKeyDown(KeyCode.UpArrow))
        {
            rb.linearVelocity = new Vector2(rb.linearVelocity.x, jumpForce);
        }
        if (move < 0)
        {
            sr.color = left;
        }
        else if(move > 0)
        {
        sr.color = right;
        }
        else
        {
            sr.color = idle;
        }
    }
}
add script to square > run

using UnityEngine;

public class scaling : MonoBehaviour
{
    public float moveForce = 30f;
    public float maxSpeed = 5f;
    private Rigidbody2D rb;
    private Vector2 input;
    void Awake()
    {
        rb = GetComponent<Rigidbody2D>();
    }
    void Update()
    {
        input.x = Input.GetAxisRaw("Horizontal");
        input.y = Input.GetAxisRaw("Vertical");
        input = input.normalized;
    }
    void FixedUpdate()
    {
        rb.AddForce(input * moveForce);
        rb.linearVelocity = Vector2.ClampMagnitude(rb.linearVelocity,maxSpeed);
    }
}

ASNCHRONOUS LOADING:

UI > Text - text mesh pro

create a script(AsyncSceneLoader):

using UnityEngine;
using UnityEngine.SceneManagement;
using System.Collections;

public class AsyncSceneLoader : MonoBehaviour
{
    void Start()
    {
        StartCoroutine(LoadSceneAsync());
    }
    IEnumerator LoadSceneAsync()
    {
        yield return new WaitForSeconds(1f);
        AsyncOperation operation =
        SceneManager.LoadSceneAsync("GameScene");
        while (!operation.isDone)
        {
            Debug.Log("Loading: " + operation.progress);
            yield return null;
        }
    }
}

In your "MainMenu" scene, change the button code to load LoadingScene:

    }
